app.get("/api/driver-applications", async (req, res) => { const applications = await DriverApplication.find().sort({ createdAt: -1 }); res.json(applications); }); app.post("/api/driver-applications/:id/approve", async (req, res) => { const updated = await DriverApplication.findByIdAndUpdate( req.params.id, { status: "approved" }, { new: true } ); res.json({ success: true, application: updated }); }); app.post("/api/driver-applications/:id/reject", async (req, res) => { const updated = await DriverApplication.findByIdAndUpdate( req.params.id, { status: "rejected" }, { new: true } ); res.json({ success: true, application: updated }); }); app.get("/api/drivers/approved-locations", async (req, res) => { const drivers = await DriverApplication.find({ status: "approved", latitude: { $ne: null }, longitude: { $ne: null } }).sort({ updatedAt: -1 }); res.json(drivers); }); app.get("/api/rides", async (req, res) => { const rides = await Ride.find().sort({ createdAt: -1 }).limit(20); res.json(rides); });
