app.get("/admin", async (req, res) => { const applications = await db.collection("drivers").find().toArray(); let html = `

Pending Driver Applications

`; applications.forEach(app => { html += `

Name: ${app.name}

Phone: ${app.phone}

Email: ${app.email}

City: ${app.city}

State: ${app.state}

Vehicle: ${app.vehicle}

Driver Type: ${app.driverType}

Status: ${app.status}

`; }); res.send(html); });const { ObjectId } = require("mongodb"); app.post("/approve-driver/:id", async (req, res) => { const id = req.params.id; await db.collection("drivers").updateOne( { _id: new ObjectId(id) }, { $set: { status: "approved" } } ); res.redirect("/admin"); }); app.post("/reject-driver/:id", async (req, res) => { const id = req.params.id; await db.collection("drivers").updateOne( { _id: new ObjectId(id) }, { $set: { status: "rejected" } } ); res.redirect("/admin"); });
