const driverId = "driver_" + Date.now(); navigator.geolocation.watchPosition(async (pos) => { const lat = pos.coords.latitude; const lng = pos.coords.longitude; await fetch("/api/driver/location", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ id: driverId, name: "Driver", lat, lng, available: true }) }); document.getElementById("status").innerText = "Tracking: " + lat + ", " + lng; });
