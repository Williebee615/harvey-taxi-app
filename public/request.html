function estimateRide(){ const pickup = document.getElementById('pickup').value const dropoff = document.getElementById('dropoff').value const service = document.getElementById('service').value fetch('/request-ride',{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({ pickup, dropoff, service }) }) document.getElementById('fare').innerHTML = "Driver is being notified..." document.getElementById('payBtn').style.display = "block" }
