async function requestRide(){ const pickup = document.getElementById("pickup").value const dropoff = document.getElementById("dropoff").value const rideType = document.getElementById("rideType").value document.getElementById("status").innerHTML = "Matching driver..." const res = await fetch("/api/rides/request",{ method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ pickupAddress:pickup, dropoffAddress:dropoff, serviceType:rideType }) }) const data = await res.json() if(data.ride.assignedDriver){ document.getElementById("driverBox").innerHTML = `

Driver Assigned

${data.ride.assignedDriver.name}
${data.ride.assignedDriver.vehicle}
Plate: ${data.ride.assignedDriver.plate}
${data.ride.assignedDriver.milesAway} miles away
Status: Driver arriving
` document.getElementById("status").innerHTML = "Driver assigned" }else{ document.getElementById("status").innerHTML = "Searching for driver..." } }
